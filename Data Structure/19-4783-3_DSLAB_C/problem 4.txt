#include <stdio.h>

#define MAX_SIZE 10  

int main()
{
    int arr[10];
    int size, i, toSearch, found;

    
    printf("Enter size of array: ");
    scanf("%d", &size);

    
    printf("Enter elements in array: ");
    for(i=0; i<size; i++)
    {
        scanf("%d", &arr[i]);
    }

    printf("\nEnter element to search: ");
    scanf("%d", &toSearch);

   
    found = 0; 
    
    for(i=0; i<size; i++)
    {
       
        if(arr[i] == toSearch)
        {
            found = 1;
            break;
        }
    }

    if(found == 1)
    {
        printf("\n%d is occure %d", toSearch, i + 1);
    }
    else
    {
        printf("\n%d is not occure the array", toSearch);
    }

    return 0;
}
